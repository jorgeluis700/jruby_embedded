  def roman
    roman = upcase
    raise unless roman?
    last = roman[-1,1]
    roman.reverse.split('').inject(0) do |result, c|
      if ROMAN_VALUES[c] < ROMAN_VALUES[last]
        result -= ROMAN_VALUES[c]
      else
        last = c
        result += ROMAN_VALUES[c]
      end
    end
  end
