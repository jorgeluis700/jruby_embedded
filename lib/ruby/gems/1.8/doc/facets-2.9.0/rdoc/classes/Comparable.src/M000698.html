  def self.[](*accessors)
    Module.new do
      include Comparable
      define_method(:comparability){ accessors }
      define_method(:<=>) do |other|
        comparability.each do |a|
          cmp = (send(a) <=> other.send(a))
          break cmp unless cmp == 0
        end
      end
    end
  end
