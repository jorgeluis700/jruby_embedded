  def purge(*trash, &block)
    trash = [nil] if trash.empty?
    r = []
    if block_given?
      each do |e|
        y = yield(e)
        r << e unless trash.any?{|t| t == y}
      end
    else
      each do |e|
        r << e unless trash.any?{|t| t == e}
      end
    end
    r
  end
