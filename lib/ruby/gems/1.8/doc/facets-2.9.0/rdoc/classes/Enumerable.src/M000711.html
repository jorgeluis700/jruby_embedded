  def count(*items, &block)
    if block || !items.empty?
      r = self
      r = r.select(&block) if block
      r = r.select{ |x| items.any?{ |i| i == x } } if !items.empty?
      r.size
    else
      begin
        size
      rescue
        i=0; each{ |e| i+=1 }; i
      end
    end
  end
