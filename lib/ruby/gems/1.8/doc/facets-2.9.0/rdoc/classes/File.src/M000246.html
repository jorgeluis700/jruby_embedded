  def self.ext(filename, new_ext=nil)
    old_ext = extname(filename)
    if new_ext == nil
      old_ext.sub(/^\./, '')
    else
      new_ext = '.' + new_ext unless (new_ext.empty? || new_ext[0,1] == '.')
      filename.chomp(old_ext) + new_ext
    end
  end
