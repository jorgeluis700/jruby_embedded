  def Continuation.create(*args, &block)
    args = [args] if not args.nil? and not args.is_a? Array # 1.6.8 compatibility
    cc = nil; result = callcc {|c| cc = c; block.call(cc) if block and args.empty?}
    result ||= args
    return *[cc, *result]
  end
